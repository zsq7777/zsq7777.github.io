	const int mHigth = 800;
	const int initRadius = 20;

	//创建图形
	initgraph(mWidth, mHigth);

	float y = initRadius;

	float vy = 1;

	float g = 1; //重力加速度
	
	bool flag = false;
	while (true)
	{
		cleardevice(); //清除
		fillcircle(mWidth / 2, y, initRadius); //在对应的坐标绘制圆
		Sleep(1);

		vy += g; //速度随着下落增加
	
		y +=  vy; //坐标随着下落方向增加

		if (y > mHigth-initRadius) { //小球落地
			vy = -vy*0.95; // 模拟回弹时速度降低
		}
		if (y >= mHigth - initRadius) {  //当位置计算>=回弹位置时
			y = mHigth - initRadius; //修正位置为触底
		}

	

	}
	
	_getch();
	closegraph();
}

              # 抛物线运动的小球

              #include <graphics.h>//图形化文件
#include <conio.h>//输入输出文件
#include <stdio.h>
#include <iostream>

int main() {

	const float mWidth = 600;
	const float mHigth = 800;
	const float initRadius = 20;

	initgraph(mWidth, mHigth);

	float y = initRadius;//y轴起始点
	float speed = 0;
	float g = 1; //重力加速度



	 
	float x = initRadius;//x轴起始点
	float xSpeed = 10; //x轴起始速度


	while (1) {
		cleardevice();
		speed = speed + g;//每下落一帧，速度增加1
		y += speed; //y轴坐标根据速度增加  -> 最后一帧如果高度为799，下落距离为30，那么y轴就会超过底部，需要修正为底部。
		if (y >= mHigth - initRadius) { //小球触底
			speed = -speed*0.9; //小球回弹 -> 回弹以后，他的回弹速度会递减，因为重力速度在增加..    *0.9 模拟回弹时速度比下落更低
		}

		if (y >= mHigth - initRadius) { //下落修正
			y = mHigth - initRadius;
		}


		x += xSpeed;
		if (x >= mWidth - initRadius) { //x轴触底
			xSpeed = -xSpeed * 0.95; //回弹速度减少   *0.9 就会停住
		}
		if (x <= initRadius) { //回到x轴原点，触底
			xSpeed = -xSpeed * 0.95; //再减少回弹速度
		}
		//std::cout <<"x轴速度" << xSpeed << std::endl;

		fillcircle(x, y, initRadius);
		Sleep(1);
	}
	
	_getch();
	closegraph();
}
