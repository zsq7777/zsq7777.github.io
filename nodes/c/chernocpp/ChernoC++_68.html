                                                                                                                                                            # 68. C++的虚析构函数

                                                                                                                                                            # 1. 虚函数

                                                                                                                                                            C++中，如果B类(子类)派生于A类(父类)，B对象当作A对象的指针处理。 那么，在调用方法时，会调用A类的方法而不是B类。

                                                                                                                                                            问：如果我想调用B类方法该怎么办？

                                                                                                                                                            答：使用虚函数。 virtual修饰父类方法，override 覆写子类方法。

                                                                                                                                                            详细使用见28.C++虚函数

                                                                                                                                                            # 2. 析构函数

                                                                                                                                                            在对象销毁时自动调用该函数。

                                                                                                                                                            26.C++析构函数

                                                                                                                                                            # 3. 析构函数+虚函数=虚析构函数

                                                                                                                                                            同1所述，B对象当作A对象的指针处理，在delete对象的时候，子类的析构函数并不会被调用。

                                                                                                                                                            问：子类析构函数不被调用会发生什么？

                                                                                                                                                            答：可能会发生内存泄漏，在子类创建一个对象数组，并在析构函数中被delete,但是，由于子类析构函数并没有被正确调用，导致了数组没被正确释放内存，导致内存泄漏。

                                                                                                                                                            问：那我应该如何使子类的析构方法被正确调用？

                                                                                                                                                            答：对父类的析构方法增加virtual关键字修饰。

                                                                                                                                                            # 代码示例

                                                                                                                                                            #include <iostream>
#include <vector>
#include <algorithm>
#include <functional>

class A {
public:
	A() {std::cout << "我是父类的构造函数" << std::endl;}
	virtual	~A() {std::cout << "我是父类的虚析构函数" << std::endl;}
};
class B : public A {
private:
	int* mArray;
public:
	B() {
		mArray = new int[5];//在堆内存上开辟了一个连续的5*4字节的int内存空间
		std::cout << "我是子类的构造函数" << std::endl; 
	}
	~B() {
		delete mArray;//释放内存
		std::cout << "我是子类的析构函数" << std::endl;
	}

};


int main() {
	A* a = new A();
	delete a;
	std::cout << "-----------" << std::endl;
	B* b = new B();
	delete b;
	std::cout << "-----------" << std::endl;
	A* c = new B();
	delete c;

}
